writing functions def function_name end function functionName() {}
array syntax; finding the index big_cities = ["SF", "LA", "NYC"]; big_cities.index("SF") var bigCities = ["SF", "LA", "NYC"]; bigCities.indexOf("SF");
key-value pairs hashes objects declared as variables; var exercise = {monday: "run", wednesday: "yoga", thursday: "climb"}; exercise.friday = "dance";
